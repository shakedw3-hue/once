# Sales Hub Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone Next.js dashboard that unifies sales data from Rosayo (WooCommerce), ALAALA (Supabase), and Once (Supabase) into one dark-themed revenue view.

**Architecture:** 3 API routes each normalize brand data into a common shape `{ grossRevenue, netRevenue, grossOrders, completedOrders, dailyData[] }`. A single dashboard page fetches all 3 and renders cards, completion table, and stacked chart.

**Tech Stack:** Next.js 15 (App Router), Tailwind CSS, shadcn/ui, Recharts, @supabase/supabase-js, date-fns

```typescript
const baseUrl = process.env.ROSAYO_WC_URL!
const key = process.env.ROSAYO_WC_KEY!
const secret = process.env.ROSAYO_WC_SECRET!

export async function fetchWooOrders(after: string, before: string) {
  const orders: any[] = []
  let page = 1

  while (true) {
    const url = new URL(`${baseUrl}/wp-json/wc/v3/orders`)
    url.searchParams.set('after', after)
    url.searchParams.set('before', before)
    url.searchParams.set('per_page', '100')
    url.searchParams.set('page', String(page))
    url.searchParams.set('consumer_key', key)
    url.searchParams.set('consumer_secret', secret)

    const res = await fetch(url.toString())
    if (!res.ok) throw new Error(`WooCommerce API error: ${res.status}`)

    const data = await res.json()
    if (data.length === 0) break

    orders.push(...data)
    if (data.length < 100) break
    page++
  }

  return orders
}
```

Create `src/app/api/rosayo/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { fetchWooOrders } from '@/lib/woocommerce'
import { BrandData, DailyEntry } from '@/lib/types'
import { format } from 'date-fns'

export async function GET(req: NextRequest) {
  const { searchParams } = req.nextUrl
  const after = searchParams.get('after')!
  const before = searchParams.get('before')!

  try {
    const orders = await fetchWooOrders(after, before)

    const dailyMap = new Map<string, DailyEntry>()

    for (const order of orders) {
      const date = format(new Date(order.date_created), 'yyyy-MM-dd')
      const amount = parseFloat(order.total) || 0
      const isCompleted = order.status === 'completed'

      const entry = dailyMap.get(date) || {
        date,
        grossRevenue: 0,
        netRevenue: 0,
        grossOrders: 0,
        completedOrders: 0,
      }

      entry.grossRevenue += amount
      entry.grossOrders += 1
      if (isCompleted) {
        entry.netRevenue += amount
        entry.completedOrders += 1
      }

      dailyMap.set(date, entry)
    }

    const daily = Array.from(dailyMap.values()).sort((a, b) => a.date.localeCompare(b.date))

    const result: BrandData = {
      brand: 'rosayo',
      grossRevenue: daily.reduce((s, d) => s + d.grossRevenue, 0),
      netRevenue: daily.reduce((s, d) => s + d.netRevenue, 0),
      grossOrders: daily.reduce((s, d) => s + d.grossOrders, 0),
      completedOrders: daily.reduce((s, d) => s + d.completedOrders, 0),
      daily,
    }

    return NextResponse.json(result)
  } catch (e: any) {
    return NextResponse.json({ error: e.message }, { status: 500 })
  }
}
```
